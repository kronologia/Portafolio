#include <stdio.h>
#include <math.h>
void main() {
    float X,L,r,h,pi;
    X=L=r=h=pi=0;
    pi=3.14;
    printf("\n");
    printf("Programa que calcule area y volumen de un cilindro\n");
    printf("Ingrese radio: ");
    scanf("%f",&r);
    printf("Ingrese altura: ");
    scanf("%f",&h);
    X=2*pi*r*(h+r);
    L=pi*pow(r,2)*h;
    printf("El area del cilindro es:%f\n",X);
    printf("El volumen del cilindro es:%f\n",L);
    getch();
}
